//@version=5
indicator("Advanced Long Term Investor Color System v2 (With Reasons)", overlay=true)

// --- UI Inputs ---
lblSizeInput = input.string("Normal", "Label Size", options=["Tiny", "Small", "Normal", "Large", "Huge"])

// Label Size Logic
tSize = size.normal
if lblSizeInput == "Tiny"
    tSize := size.tiny
if lblSizeInput == "Small"
    tSize := size.small
if lblSizeInput == "Large"
    tSize := size.large
if lblSizeInput == "Huge"
    tSize := size.huge

// Indicators
ema50  = ta.ema(close, 50)
ema200 = ta.ema(close, 200)
rsi    = ta.rsi(close, 14)
volAvg = ta.sma(volume, 20)

// Conditions
priceAbove200 = close > ema200
goldenTrend   = ema50 > ema200
strongVol     = volume > volAvg
healthyRSI    = rsi >= 45 and rsi <= 65
overHeat      = rsi > 70
bearTrend     = close < ema200 and ema50 < ema200
breakout      = close > ta.highest(high, 50)[1]

// Text holders
string status = ""
string reason = ""
string explain = ""
color bg = na

// ZONE LOGIC WITH REASONS
if priceAbove200 and goldenTrend and healthyRSI and strongVol
    status  := "🟢 STRONG BUY ZONE"
    reason  := "Price > 200 EMA | Golden Trend | RSI Healthy | Volume Strong"
    explain := "Best long-term buying zone"
    bg := color.new(color.green, 80)

else if priceAbove200 and goldenTrend
    status  := "🟩 BUY / ACCUMULATE ZONE"
    reason  := "Price above 200 EMA | Trend Positive"
    explain := "Safe zone, buy slowly on dips"
    bg := color.new(color.lime, 80)

else if breakout and strongVol and rsi > 60
    status  := "🔵 BREAKOUT ZONE"
    reason  := "New High Breakout | High Volume | Strong Momentum"
    explain := "Fast growth phase – medium risk"
    bg := color.new(color.blue, 80)

else if priceAbove200
    status  := "🟡 HOLD / WAIT ZONE"
    reason  := "Price above 200 EMA but momentum weak"
    explain := "Hold existing, wait for clear signal"
    bg := color.new(color.yellow, 80)

else if overHeat
    status  := "🟣 PROFIT BOOKING ZONE"
    reason  := "RSI above 70 | Stock overheated"
    explain := "Book partial profit, avoid fresh buy"
    bg := color.new(color.purple, 80)

else if bearTrend
    status  := "🔴 AVOID / EXIT ZONE"
    reason  := "Price below 200 EMA | 50 EMA below 200 EMA"
    explain := "Long-term trend bearish – protect capital"
    bg := color.new(color.red, 80)

else
    status  := "⚪ NO CLEAR SIGNAL"
    reason  := "Trend mixed | Waiting for confirmation"
    explain := "Better to wait – no action now"
    bg := color.new(color.gray, 85)

// Plot EMAs
plot(ema50, color=color.blue, title="EMA 50")
plot(ema200, color=color.red, title="EMA 200")

// Show Detailed Label (only last candle)
if barstate.islast
    label.new(bar_index, high, 
      status + "\n" + 
      "Reason: " + reason + "\n" + 
      "Note: " + explain + "\n" +
      "------------------------" + "\n" +
      "EMA 50: " + str.tostring(ema50, format.mintick) + "\n" +
      "EMA 200: " + str.tostring(ema200, format.mintick),
      style=label.style_label_down,
      color=color.black,
      textcolor=color.white,
      size=tSize) // Using the dynamic size variable here

// Background color
bgcolor(bg)
